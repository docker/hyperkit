These are binaries from
    http://tinycorelinux.net
with the following patch applied:

Source is available http://www.tinycorelinux.net/6.x/x86/release/src/

curl -O http://www.tinycorelinux.net/6.x/x86/release/distribution_files/vmlinuz64
# c4f46b5cd826d7a765a3e85439309cfc  vmlinuz64
mv vmlinuz64 vmlinuz
curl -O http://www.tinycorelinux.net/6.x/x86/release/distribution_files/core.gz
mv core.gz initrd.gz
mkdir initrd
( cd initrd ; gzcat ../initrd.gz | sudo cpio -idm )
sudo sed -i -e '/^# ttyS0$/s#^..##' initrd/etc/securetty 
sudo sed -i -e '/^tty1:/s#tty1#ttyS0#g' initrd/etc/inittab
( cd initrd ; find | sudo cpio -o -H newc ) | gzip -c > initrd.gz
